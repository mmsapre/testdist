package com.acme.graavl.steps;

import com.acme.graavl.config.ConnectorConfig;
import com.acme.graavl.config.WhereConfig;
import com.acme.graavl.core.PipelineRecord;
import com.acme.graavl.dsl.DslSqlBuilder;
import com.acme.graavl.dsl.NamedParameterStatement;
import org.jeasy.batch.core.processor.RecordProcessor;
import org.jeasy.batch.core.record.Record;

import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.ResultSet;
import java.sql.ResultSetMetaData;
import java.util.*;

public class GraavlJdbcProcessor implements RecordProcessor<PipelineRecord, PipelineRecord> {

    private final ConnectorConfig connectorConfig;
    private final DataSource dataSource;
    private final DslSqlBuilder builder = new DslSqlBuilder();

    public GraavlJdbcProcessor(ConnectorConfig connectorConfig, DataSource dataSource) {
        this.connectorConfig = connectorConfig;
        this.dataSource = dataSource;
    }

    @Override
    public PipelineRecord processRecord(Record<PipelineRecord> record) throws Exception {
        PipelineRecord r = record.getPayload();

        String sql = builder.build(connectorConfig);
        Map<String, Object> params = resolveParams(r.getEvent());

        List<Map<String, Object>> rows = new ArrayList<>();

        try (Connection conn = dataSource.getConnection();
             NamedParameterStatement stmt = new NamedParameterStatement(conn, sql);
             ResultSet rs = execute(stmt, params)) {

            ResultSetMetaData meta = rs.getMetaData();

            while (rs.next()) {
                Map<String, Object> row = new LinkedHashMap<>();
                for (int i = 1; i <= meta.getColumnCount(); i++) {
                    row.put(meta.getColumnLabel(i), rs.getObject(i));
                }
                rows.add(row);
            }
        }

        r.getQueryResult().put("sql", sql);
        r.getQueryResult().put("rows", rows);
        return r;
    }

    private ResultSet execute(NamedParameterStatement stmt, Map<String, Object> params) throws Exception {
        for (Map.Entry<String, Object> e : params.entrySet()) {
            stmt.setObject(e.getKey(), e.getValue());
        }
        return stmt.executeQuery();
    }

    private Map<String, Object> resolveParams(Map<String, Object> event) {
        Map<String, Object> params = new LinkedHashMap<>();

        for (WhereConfig w : connectorConfig.getWhere()) {
            Object value = null;

            if ("event".equalsIgnoreCase(w.getSource())) {
                value = event.get(w.getKey());
            } else if ("static".equalsIgnoreCase(w.getSource())) {
                value = w.getValue();
            } else if ("system".equalsIgnoreCase(w.getSource())) {
                value = System.getProperty(w.getKey());
            } else if ("env".equalsIgnoreCase(w.getSource())) {
                value = System.getenv(w.getKey());
            }

            params.put(w.getParam(), value);
        }

        return params;
    }
}

package com.acme.graavl.dsl;

import java.sql.*;
import java.util.*;

public class NamedParameterStatement implements AutoCloseable {

    private final PreparedStatement statement;
    private final Map<String, List<Integer>> indexMap = new HashMap<>();

    public NamedParameterStatement(Connection connection, String sql) throws SQLException {
        String parsed = parse(sql, indexMap);
        this.statement = connection.prepareStatement(parsed);
    }

    public void setObject(String name, Object value) throws SQLException {
        List<Integer> indexes = indexMap.get(name);
        if (indexes == null) {
            return;
        }

        for (Integer i : indexes) {
            statement.setObject(i, value);
        }
    }

    public ResultSet executeQuery() throws SQLException {
        return statement.executeQuery();
    }

    private static String parse(String query, Map<String, List<Integer>> indexMap) {
        StringBuilder parsed = new StringBuilder();
        int length = query.length();
        int index = 1;

        for (int i = 0; i < length; i++) {
            char c = query.charAt(i);
            if (c == ':') {
                int j = i + 1;
                while (j < length && Character.isJavaIdentifierPart(query.charAt(j))) {
                    j++;
                }
                String name = query.substring(i + 1, j);
                parsed.append('?');
                indexMap.computeIfAbsent(name, k -> new ArrayList<>()).add(index++);
                i = j - 1;
            } else {
                parsed.append(c);
            }
        }
        return parsed.toString();
    }

    @Override
    public void close() throws SQLException {
        statement.close();
    }
}

package com.acme.graavl.policy;

import com.acme.graavl.dsl.NamedParameterStatement;

import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

public class PolicyQueryExecutor {

    private final DataSource dataSource;

    public PolicyQueryExecutor(DataSource dataSource) {
        this.dataSource = dataSource;
    }

    public List<PolicyResult> execute(String sql, Map<String, Object> params) throws Exception {
        List<PolicyResult> out = new ArrayList<>();

        try (Connection conn = dataSource.getConnection();
             NamedParameterStatement stmt = new NamedParameterStatement(conn, sql)) {

            for (Map.Entry<String, Object> e : params.entrySet()) {
                stmt.setObject(e.getKey(), e.getValue());
            }

            try (ResultSet rs = stmt.executeQuery()) {
                while (rs.next()) {
                    PolicyResult pr = new PolicyResult();
                    pr.setPolicyKey(rs.getString("policyKey"));
                    pr.setAction(PolicyAction.fromDb(rs.getString("action")));
                    pr.setPriority(rs.getInt("priority"));
                    try {
                        pr.setJsonLogic(rs.getString("jsonLogic"));
                    } catch (Exception ignored) {
                    }
                    out.add(pr);
                }
            }
        }

        return out;
    }
}

package expressions;

import com.brunocesar.jsonlogic.*;

import dsl.QueryDslExecutor;
import model.PayloadContext;

import java.util.List;
import java.util.Map;

public class EnrichExpression implements JsonLogicExpression {

    private QueryDslExecutor executor;

    public EnrichExpression(QueryDslExecutor executor){
        this.executor = executor;
    }

    @Override
    public String key() {
        return "enrich";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           List<JsonLogicNode> args,
                           Object data,
                           String path) throws Exception {

        PayloadContext payload = (PayloadContext)data;

        String alias =
                args.get(0).toString();

        Map<String,Object> enrich =
                executor.lookup(alias,payload);

        return enrich;
    }
}

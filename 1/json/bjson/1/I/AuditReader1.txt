package com.hsbc.gbm.hss.mrd.distribution.audit;

import org.springframework.jdbc.core.JdbcTemplate;

import java.util.List;
import java.util.Map;

public class AuditReader {

    private final JdbcTemplate jdbcTemplate;
    private final String table;

    public AuditReader(JdbcTemplate jdbcTemplate, String table) {
        this.jdbcTemplate = jdbcTemplate;
        this.table = table;
    }

    public List<Map<String,Object>> findPending(String dataKey) {

        String sql = """
            SELECT *
            FROM %s
            WHERE data_key = ?
              AND state = 'PENDING'
            ORDER BY created_at
        """.formatted(table);

        return jdbcTemplate.queryForList(sql, dataKey);
    }

    public List<Map<String,Object>> findByLoadRowId(Long loadRowId) {

        String sql = """
            SELECT *
            FROM %s
            WHERE load_row_id = ?
        """.formatted(table);

        return jdbcTemplate.queryForList(sql, loadRowId);
    }

    public List<Map<String,Object>> findRecent(int limit) {

        String sql = """
            SELECT *
            FROM %s
            ORDER BY created_at DESC
            LIMIT ?
        """.formatted(table);

        return jdbcTemplate.queryForList(sql, limit);
    }

    String sql = """

package com.acme.graavl.logic.extensions;

import io.github.brunocribeiro.jsonlogic.JsonLogicOperation;

public class CaseExpression implements JsonLogicOperation {

    @Override
    public Object apply(Object[] args) {

        for(int i=0;i<args.length;i+=2) {

            if(Boolean.TRUE.equals(args[i]))
                return args[i+1];
        }

        return null;
    }
}

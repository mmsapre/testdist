public class CaseExpression implements JsonLogicExpression {

    @Override
    public String key() {
        return "case";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           JsonLogicArray args,
                           Object data) {

        for (int i = 0; i < args.size(); i++) {

            Object entry = args.get(i);

            if (!(entry instanceof Map)) continue;

            Map<?,?> m = (Map<?,?>) entry;

            if (m.containsKey("when")) {

                Object cond = evaluator.evaluate(m.get("when"), data);

                if (JsonLogic.truthy(cond)) {
                    return evaluator.evaluate(m.get("then"), data);
                }
            }

            if (m.containsKey("else")) {
                return evaluator.evaluate(m.get("else"), data);
            }
        }

        return null;
    }
}

public class ContextExpression implements JsonLogicExpression {

    @Override
    public String key() {
        return "ctx";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           JsonLogicArray args,
                           Object data) {

        Map<String,Object> root = (Map<String,Object>) data;
        Map<String,Object> ctx = (Map<String,Object>) root.get("ctx");

        String key = evaluator.evaluate(args.get(0), data).toString();

        return resolve(ctx, key);
    }

    private Object resolve(Map<String,Object> ctx, String key) {

        String[] parts = key.split("\\.");

        Object current = ctx;

        for (String p : parts) {
            if (!(current instanceof Map)) return null;
            current = ((Map<?,?>) current).get(p);
        }

        return current;
    }
}

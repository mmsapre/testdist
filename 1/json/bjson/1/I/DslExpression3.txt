package expressions;

import com.brunocesar.jsonlogic.*;

import dsl.QueryDslExecutor;

public class DslExpression implements JsonLogicExpression {

    private QueryDslExecutor executor;

    public DslExpression(QueryDslExecutor executor) {
        this.executor = executor;
    }

    @Override
    public String key() {
        return "dsl";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           JsonLogicArray args,
                           Object data)
            throws JsonLogicEvaluationException {

        try {

            // alias can be dynamic
            Object aliasObj =
                    evaluator.evaluate(args.get(0), data);

            if (aliasObj == null) {
                return null;
            }

            String alias = aliasObj.toString();

            Object result =
                    executor.execute(alias, data);

            // IMPORTANT:
            // return primitive / list / map directly
            return result;

        } catch (Exception e) {
            throw new JsonLogicEvaluationException(
                    "DSL execution failed", e);
        }
    }
}

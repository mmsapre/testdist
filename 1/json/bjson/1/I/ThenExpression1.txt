package expressions;

import com.brunocesar.jsonlogic.*;

public class ThenExpression implements JsonLogicExpression {

    @Override
    public String key() {
        return "then";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           JsonLogicArray args,
                           Object data)
            throws JsonLogicEvaluationException {

        return evaluator.evaluate(args.get(0), data);
    }
}

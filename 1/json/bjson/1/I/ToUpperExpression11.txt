public class ToUpperExpression implements JsonLogicExpression {

    @Override
    public String key() {
        return "upper";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           JsonLogicArray args,
                           Object data) {

        Object val = evaluator.evaluate(args.get(0), data);

        return val == null ? null : val.toString().toUpperCase();
    }
}

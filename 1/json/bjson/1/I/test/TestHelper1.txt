class TestHelper {

    static JsonLogic build(JsonLogicExpression... expressions) {

        JsonLogic logic = new JsonLogic();

        for (JsonLogicExpression e : expressions) {
            logic.addOperation(e);
        }

        return logic;
    }

    static Map<String,Object> data() {
        Map<String,Object> ctx = new HashMap<>();
        Map<String,Object> data = new HashMap<>();
        data.put("ctx", ctx);
        return data;
    }
}

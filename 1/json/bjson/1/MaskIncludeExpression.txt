package expressions;

import com.brunocesar.jsonlogic.*;
import com.jayway.jsonpath.*;

import mask.MaskPathIndex;
import model.PayloadContext;

import java.util.List;

public class MaskIncludeExpression implements JsonLogicExpression {

    private MaskPathIndex index;

    public MaskIncludeExpression(MaskPathIndex index){
        this.index = index;
    }

    @Override
    public String key() {
        return "mask_include";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           List<JsonLogicNode> args,
                           Object data,
                           String path) {

        PayloadContext payload = (PayloadContext)data;

        DocumentContext ctx =
                JsonPath.parse(payload.getJson());

        for(JsonPath p:index.getInclude("PARTIAL_MASK"))
            ctx.set(p.getPath(),"****");

        payload.setJson(ctx.jsonString());

        return payload;
    }
}

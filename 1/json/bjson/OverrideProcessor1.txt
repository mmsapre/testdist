package processors;

import engine.ActionProcessor;
import model.*;

import com.jayway.jsonpath.*;

import java.util.Map;

public class OverrideProcessor implements ActionProcessor {

    public PayloadContext process(PayloadContext payload,
                                  ActionResult result){

        Map<String,Object> overrides =
                result.getParameters();

        DocumentContext ctx =
                JsonPath.parse(payload.getJson());

        for(String p:overrides.keySet())
            ctx.set(p,overrides.get(p));

        payload.setJson(ctx.jsonString());

        return payload;
    }
}

package operators;

import com.brunocesar.jsonlogic.*;

import java.util.List;
import java.util.Objects;

public class SwitchExpression implements JsonLogicExpression {

    @Override
    public String key() {
        return "switch";
    }

    @Override
    public Object evaluate(JsonLogicEvaluator evaluator,
                           List<JsonLogicNode> args,
                           Object data,
                           String path) throws Exception {

        if(args.size() < 3)
            throw new IllegalArgumentException("switch requires at least 3 arguments");

        Object value =
            args.get(0).evaluate(evaluator,data,path);

        int i = 1;

        while(i < args.size()-1){

            Object caseVal =
                args.get(i).evaluate(evaluator,data,path);

            if(Objects.equals(value,caseVal)){

                return args.get(i+1)
                        .evaluate(evaluator,data,path);
            }

            i += 2;
        }

        return args.get(args.size()-1)
                .evaluate(evaluator,data,path);
    }
}

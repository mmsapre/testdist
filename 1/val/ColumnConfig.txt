public class ColumnConfig {

    private String expr;

    private String alias;

    public String getExpr() {
        return expr;
    }

    public String getAlias() {
        return alias;
    }

    public void setExpr(String expr) {
        this.expr = expr;
    }

    public void setAlias(String alias) {
        this.alias = alias;
    }
}

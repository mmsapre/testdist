package com.acme.graavl.config;

public class JdbcReaderConfig {

    private String datasource;
    private SqlConfig sql;


    public String payloadColumn;         // optional
    public List<String> jsonColumns;     // optional
    public List<String> xmlColumns;      // optional
}
    public String getDatasource() {
        return datasource;
    }

    public void setDatasource(String datasource) {
        this.datasource = datasource;
    }

    public SqlConfig getSql() {
        return sql;
    }

    public void setSql(SqlConfig sql) {
        this.sql = sql;
    }
}

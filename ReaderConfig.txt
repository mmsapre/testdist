package com.acme.graavl.config;

public class ReaderConfig {

    /**
     * jdbc | api | file | jms (future)
     */
    private String type;

    private JdbcReaderConfig jdbc;

    public String getType() {
        return type;
    }

    public void setType(String type) {
        this.type = type;
    }

    public JdbcReaderConfig getJdbc() {
        return jdbc;
    }

    public void setJdbc(JdbcReaderConfig jdbc) {
        this.jdbc = jdbc;
    }
}

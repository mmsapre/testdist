package com.acme.graavl.config;

import java.util.Map;

public class ValidationConfig {

    private boolean enabled = true;
    private boolean bypass = false;
    private String strategy = "PRIORITY_HIGHEST_WINS";
    private Map<String, PolicyConfig> policies;

    public ValidationConfig() {}

    public boolean isEnabled() {
        return enabled;
    }

    public void setEnabled(boolean enabled) {
        this.enabled = enabled;
    }

    public boolean isBypass() {
        return bypass;
    }

    public void setBypass(boolean bypass) {
        this.bypass = bypass;
    }

    public String getStrategy() {
        return strategy;
    }

    public void setStrategy(String strategy) {
        this.strategy = strategy;
    }

    public Map<String, PolicyConfig> getPolicies() {
        return policies;
    }

    public void setPolicies(Map<String, PolicyConfig> policies) {
        this.policies = policies;
    }
}
